local function getClosestTarget()
    local closestPlayer = nil
    local shortestDistance = math.huge
    local camera = workspace.CurrentCamera
    local center = fovPosition == "Center" and camera.ViewportSize / 2 or UserInputService:GetMouseLocation()
    
    for _, player in pairs(Players:GetPlayers()) do
        if player ~= LocalPlayer and player.Character and player.Character:FindFirstChildOfClass("Humanoid") and player.Character:FindFirstChildOfClass("Humanoid").Health > 0 then
            
            local targetPart = player.Character:FindFirstChild(aimbotTargetPart) or player.Character:FindFirstChild("HumanoidRootPart")
            if not targetPart then continue end

            if aimbotTeamCheck and player.Team == LocalPlayer.Team then continue end
            
            local isFriend = false
            pcall(function() isFriend = LocalPlayer:IsFriendsWith(player.UserId) end)
            if aimbotFriendCheck and isFriend then continue end
            
            local pos, onScreen = camera:WorldToViewportPoint(targetPart.Position)
            
            if onScreen then
                if wallCheckEnabled then
                    local origin = camera.CFrame.Position
                    local dir = targetPart.Position - origin
                    local raycastParams = RaycastParams.new()
                    raycastParams.FilterType = Enum.RaycastFilterType.Exclude
                    raycastParams.FilterDescendantsInstances = {LocalPlayer.Character, player.Character, camera}
                    local res = workspace:Raycast(origin, dir, raycastParams)
                    if res then continue end
                end
                
                local distance = (Vector2.new(pos.X, pos.Y) - center).Magnitude
                if distance < fovRadius and distance < shortestDistance then
                    closestPlayer = player
                    shortestDistance = distance
                end
            end
        end
    end
    return closestPlayer
end

UserInputService.InputBegan:Connect(function(input, gameProcessed)
    if gameProcessed then return end
    
    if ctrlClickTpEnabled and input.UserInputType == Enum.UserInputType.MouseButton1 and UserInputService:IsKeyDown(Enum.KeyCode.LeftControl) then
        local mouse = LocalPlayer:GetMouse()
        local character = LocalPlayer.Character
        if character and character:FindFirstChild("HumanoidRootPart") and mouse.Hit then
            character.HumanoidRootPart.CFrame = CFrame.new(mouse.Hit.Position + Vector3.new(0, 3, 0))
        end
    end

    if dashEnabled and input.KeyCode == dashKeybind then
        local char = LocalPlayer.Character
        local hrp = char and char:FindFirstChild("HumanoidRootPart")
        if hrp then
            hrp.CFrame = hrp.CFrame * CFrame.new(0, 0, -dashDistance)
        end
    end
    
    local matches = false
    if aimbotKeyType == "Keyboard" and input.KeyCode == aimbotKeybind then
        matches = true
    elseif aimbotKeyType == "Mouse" and input.UserInputType == aimbotKeybind then
        matches = true
    end

    if matches then
        if aimbotMode == "Toggle" then
            isAiming = not isAiming
            if not isAiming then
                aimbotTarget = nil
            end
        else
            isAiming = true
        end
    end
end)

RunService.Heartbeat:Connect(function(dt)
    local character = LocalPlayer.Character
    local hrp = character and character:FindFirstChild("HumanoidRootPart")
    local humanoid = character and character:FindFirstChildOfClass("Humanoid")
    local camera = workspace.CurrentCamera

    if humanoid then
        if not walkSpeedEnabled then
            originalWalkSpeed = humanoid.WalkSpeed
        end
        if not jumpPowerEnabled then
            originalJumpPower = humanoid.JumpPower
        end
        if noJumpCooldown then
            humanoid.JumpPower = jumpPowerEnabled and jumpPowerValue or originalJumpPower
        end
    end
    
    if autoClickerEnabled and UserInputService:IsMouseButtonPressed(Enum.UserInputType.MouseButton1) then
        if mouse1click then
            mouse1click()
        else
            VirtualUser:ClickButton1(Vector2.new(0, 0))
        end
    end

    if triggerbotEnabled then
        local mouse = LocalPlayer:GetMouse()
        local target = mouse.Target
        if target and target.Parent then
            local char = target.Parent
            if char:IsA("Accessory") then char = char.Parent end
            local player = Players:GetPlayerFromCharacter(char)
            if player and player ~= LocalPlayer then
                local hum = char:FindFirstChildOfClass("Humanoid")
                if hum and hum.Health > 0 then
                    local passCheck = true
                    if triggerbotTeamCheck and player.Team == LocalPlayer.Team then passCheck = false end
                    local isFriend = false
                    pcall(function() isFriend = LocalPlayer:IsFriendsWith(player.UserId) end)
                    if triggerbotFriendCheck and isFriend then passCheck = false end
                    
                    if passCheck then
                        if mouse1click then
                            mouse1click()
                        else
                            VirtualUser:ClickButton1(Vector2.new(0, 0))
                        end
                    end
                end
            end
        end
    end

    if antiAimEnabled and hrp and humanoid and humanoid.Health > 0 then
        if antiAimMode == "Backwards" then
            local lookAt = camera.CFrame.LookVector
            local flatLook = Vector3.new(lookAt.X, 0, lookAt.Z).Unit
            hrp.CFrame = CFrame.lookAt(hrp.Position, hrp.Position - flatLook)
        elseif antiAimMode == "Jitter" then
            local lookAt = camera.CFrame.LookVector
            local flatLook = Vector3.new(lookAt.X, 0, lookAt.Z).Unit
            local jitterAngle = math.random() > 0.5 and math.rad(90) or math.rad(-90)
            hrp.CFrame = CFrame.lookAt(hrp.Position, hrp.Position + flatLook) * CFrame.Angles(0, jitterAngle, 0)
        elseif antiAimMode == "Spin" then
            hrp.CFrame = hrp.CFrame * CFrame.Angles(0, math.rad(30), 0)
        end
    end

    if thirdPersonEnabled then
        LocalPlayer.CameraMode = Enum.CameraMode.Classic
        LocalPlayer.CameraMinZoomDistance = thirdPersonDistance
        LocalPlayer.CameraMaxZoomDistance = thirdPersonDistance
    end

    if flyEnabled and hrp and humanoid then
        local moveDir = humanoid.MoveDirection
        local flyDir = Vector3.new(0, 0, 0)

        if moveDir.Magnitude > 0 then
            local rawDir = camera.CFrame:VectorToObjectSpace(moveDir)
            flyDir = (camera.CFrame.LookVector * -rawDir.Z) + (camera.CFrame.RightVector * rawDir.X)
        end

        if UserInputService:IsKeyDown(Enum.KeyCode.Space) or humanoid.Jump then
            flyDir = flyDir + Vector3.new(0, 1, 0)
        elseif UserInputService:IsKeyDown(Enum.KeyCode.LeftShift) then
            flyDir = flyDir + Vector3.new(0, -1, 0)
        end

        local currentFlyMethod = flyMethod
        local targetPart = hrp
        
        if currentFlyMethod == "Car Fly" then
            if humanoid.SeatPart and humanoid.SeatPart:IsA("VehicleSeat") then
                targetPart = humanoid.SeatPart
            else
                currentFlyMethod = "Normal"
            end
        end

        if lastTargetPart and lastTargetPart ~= targetPart then
            if bv then bv:Destroy() bv = nil end
            if bg then bg:Destroy() bg = nil end
        end
        lastTargetPart = targetPart

        if currentFlyMethod == "Car Fly" or currentFlyMethod == "Normal" then
            if currentFlyMethod == "Normal" then
                humanoid.PlatformStand = true
            else
                targetPart.AssemblyLinearVelocity = Vector3.new(0, 0, 0)
            end
            
            if not bv or bv.Parent ~= targetPart then
                if bv then bv:Destroy() end
                bv = Instance.new("BodyVelocity")
                bv.MaxForce = Vector3.new(1e5, 1e5, 1e5)
                bv.Parent = targetPart
            end
            if not bg or bg.Parent ~= targetPart then
                if bg then bg:Destroy() end
                bg = Instance.new("BodyGyro")
                bg.MaxTorque = Vector3.new(1e5, 1e5, 1e5)
                bg.Parent = targetPart
            end
            bg.CFrame = camera.CFrame
            if flyDir.Magnitude > 0 then
                bv.Velocity = flyDir.Unit * flySpeed
            else
                bv.Velocity = Vector3.new(0, 0, 0)
            end
        elseif currentFlyMethod == "CFrame" then
            humanoid.PlatformStand = true
            if bv then bv:Destroy() bv = nil end
            if bg then bg:Destroy() bg = nil end
            targetPart.AssemblyLinearVelocity = Vector3.new(0, 0, 0)
            if flyDir.Magnitude > 0 then
                targetPart.CFrame = targetPart.CFrame + (flyDir.Unit * flySpeed * dt)
            else
                local verticalVel = Vector3.new(0, 0, 0)
                if UserInputService:IsKeyDown(Enum.KeyCode.Space) or humanoid.Jump then
                    verticalVel = Vector3.new(0, flySpeed, 0)
                elseif UserInputService:IsKeyDown(Enum.KeyCode.LeftShift) then
                    verticalVel = Vector3.new(0, -flySpeed, 0)
                end
                targetPart.CFrame = targetPart.CFrame + (verticalVel * dt)
            end
        end
    else
        if bv then bv:Destroy() bv = nil end
        if bg then bg:Destroy() bg = nil end
        if lastTargetPart then lastTargetPart = nil end
        if humanoid and humanoid.PlatformStand then
            humanoid.PlatformStand = false
        end
    end

    if humanoid and hrp then
        if walkSpeedEnabled then
            if speedMethod == "WalkSpeed" then
                humanoid.WalkSpeed = walkSpeedValue
            elseif speedMethod == "CFrame" then
                humanoid.WalkSpeed = originalWalkSpeed
                if humanoid.MoveDirection.Magnitude > 0 then
                    hrp.CFrame = hrp.CFrame + (humanoid.MoveDirection * (walkSpeedValue * 0.1) * dt * 60)
                end
            end
        else
            if humanoid.WalkSpeed ~= originalWalkSpeed then
                humanoid.WalkSpeed = originalWalkSpeed
            end
        end

        if jumpPowerEnabled then 
            humanoid.JumpPower = jumpPowerValue 
            humanoid.UseJumpPower = true
        else
            if humanoid.JumpPower ~= originalJumpPower then
                humanoid.JumpPower = originalJumpPower
                humanoid.UseJumpPower = false
            end
        end

        if bhopEnabled and UserInputService:IsKeyDown(Enum.KeyCode.Space) and humanoid.FloorMaterial ~= Enum.Material.Air then
            humanoid:ChangeState(Enum.HumanoidStateType.Jumping)
        end

        if spinbotEnabled then
            hrp.CFrame = hrp.CFrame * CFrame.Angles(0, math.rad(spinbotSpeed), 0)
        end

        if noSitEnabled then
            humanoid.Sit = false
        end
    end

    if fovEnabled and camera then
        camera.FieldOfView = fovValue
    elseif camera then
        camera.FieldOfView = 70
    end

    for _, player in pairs(Players:GetPlayers()) do
        if player ~= LocalPlayer and player.Character then
            local char = player.Character
            local hum = char:FindFirstChildOfClass("Humanoid")
            
            local isFriend = false
            pcall(function() isFriend = LocalPlayer:IsFriendsWith(player.UserId) end)
            
            local passCheck = true
            if hitboxTeamCheck and player.Team == LocalPlayer.Team then passCheck = false end
            if hitboxFriendCheck and isFriend then passCheck = false end
            
            if hitboxEnabled and passCheck and hum and hum.Health > 0 then
                local part = char:FindFirstChild(hitboxPart)
                if part and part:IsA("BasePart") then
                    if not hitboxCache[part] then
                        hitboxCache[part] = {
                            Size = part.Size,
                            Material = part.Material,
                            Transparency = part.Transparency,
                            Massless = part.Massless,
                            CanCollide = part.CanCollide
                        }
                    end
                    part.Size = Vector3.new(hitboxSize, hitboxSize, hitboxSize)
                    part.Transparency = hitboxTransparency
                    part.Color = hitboxColor
                    pcall(function()
                        part.Material = Enum.Material[hitboxMaterial]
                    end)
                    if hitboxPart == "Head" then
                        part.Massless = true
                    end
                    part.CanCollide = false
                end
            end

            if chamsEnabled and passCheck and hum and hum.Health > 0 then
                local highlight = char:FindFirstChild("LawChams")
                if not highlight then
                    highlight = Instance.new("Highlight")
                    highlight.Name = "LawChams"
                    highlight.Parent = char
                end
                highlight.FillColor = chamsFillColor
                highlight.OutlineColor = chamsOutlineColor
                highlight.FillTransparency = 0.5
                highlight.OutlineTransparency = 0
            else
                local highlight = char:FindFirstChild("LawChams")
                if highlight then
                    highlight:Destroy()
                end
            end
        end
    end

    for part, original in pairs(hitboxCache) do
        if part and part.Parent then
            local char = part.Parent
            local player = Players:GetPlayerFromCharacter(char)
            local isFriend = false
            if player then
                pcall(function() isFriend = LocalPlayer:IsFriendsWith(player.UserId) end)
            end
            
            local passCheck = true
            if player then
                if hitboxTeamCheck and player.Team == LocalPlayer.Team then passCheck = false end
                if hitboxFriendCheck and isFriend then passCheck = false end
            end
            
            local currentTargetPart = char and char:FindFirstChild(hitboxPart)
            
            if not hitboxEnabled or not passCheck or currentTargetPart ~= part or (char:FindFirstChildOfClass("Humanoid") and char:FindFirstChildOfClass("Humanoid").Health <= 0) then
                part.Size = original.Size
                part.Material = original.Material
                part.Transparency = original.Transparency
                part.Massless = original.Massless
                part.CanCollide = original.CanCollide
                hitboxCache[part] = nil
            end
        else
            hitboxCache[part] = nil
        end
    end
end)
